# x es la edad en aos cumplidos
# y es el ao actual 
# qx es la tabla de mortalidad (matriz por edad y ao)
# w es la edad final de la tabla

ax <- function(x, y, qx, w){
  
  anos_max <- w-x
  v <- 1/(1+0.05)
  kpx <- 1
  evp <- v * qx[x+1, y-1998]
  
  for(j in 2:anos_max){
    
    if(y-1998+(j-2) < 152){
      
      kpx <- kpx * (1-qx[x+(j-1), y-1998+(j-2)])
      kqx <- qx[x+j, y-1998+j-1]
      vp <- v^(j)
      evp <- evp + vp * (kpx * kqx)
      
    }  
    else{
      
      kpx <- kpx * (1-qx[x+(j-1), 152])
      kqx <- qx[x+j, 152]
      vp <- v^(j)
      evp <- evp + vp * (kpx * kqx)
      
    }
    
  }
  
  return(as.numeric(evp[[1]]))

}

#Funcin que calcula la probabilidad de que una persona de x aos sobreviva n aos.
nPx <- function(x,y,n,qx,w){
  
  npx <- 1
  if(n>=1){
    
    for (t in 0:(n-1)) {
      #print(as.numeric(npx))
      #print((as.numeric(qx[x+1+t, y-1998+t])))
      
      npx <- npx * (1 - qx[x+1+t, y-1998+t])
      
      #print(as.numeric(npx))
      #print("vue")
    
    }
  }

  #print("resultado:")
  return(as.numeric(npx))
}

Axn <- function(x,y,n,qx,w){
#' Axn - Calcular la esperanza del valor presente de un seguro temporal
#'
#' La funcin Axn calcula la esperanza del valor presente de un seguro temporal
#' cuyo beneficio se paga al final del ao de muerte, utilizando la tabla de mortalidad dinmica de la SUPEN.
#'
#' @param x Edad en aos cumplidos del asegurado.
#' @param y Ao actual.
#' @param n Temporalidad del seguro, es decir, el nmero de aos para los cuales se est contratando el seguro.
#' @param qx Tabla de mortalidad, representada como una matriz por edad y ao.
#' @param w Edad final de la tabla de mortalidad.
#' @return La esperanza del valor presente de un seguro temporal.
  
  if(n < 1){
    stop("La temporalidad del seguro debe ser de al menos un ao")
  }
  
  if(n> x+w){
    stop(paste("La temporalidad debe ser como mximo de: ", x+w))
  } 
  
  if(x<0 || x>w){
    stop("Edad en aos cumplidos invlida")
  }
  
  if(y<2000 || y >2150){
    stop("Ao actual invlido")
  }
  
  
  #iniciar variables
  axn <- 0
  v <- (1/(1 + 0.05))
  #xqx <- 0
  
  for (k in 0:(n-1)){
    
    dife_kqx <- nPx(x,y,k,qx,w) - nPx(x,y,k+1,qx,w)
    #print(dife_kqx)
    axn <- axn + v^{k+1} * dife_kqx 
    #print(paste("axn: ", axn))
  }

  return(axn)   
}
